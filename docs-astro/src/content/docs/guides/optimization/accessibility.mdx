# Accessibility

Build inclusive e-commerce experiences that work for all users, including those using assistive technologies like screen readers, voice control, and keyboard navigation.

## Accessibility Overview

The Next Commerce SDK provides the foundation for accessible e-commerce implementations, but proper HTML structure, ARIA labels, and interaction patterns are essential for full accessibility compliance.

### Key Accessibility Principles

- **Perceivable** - Information must be presentable in ways users can perceive
- **Operable** - Interface components must be operable by all users
- **Understandable** - Information and UI operation must be understandable
- **Robust** - Content must be robust enough for various assistive technologies

## ARIA Implementation

### Dynamic Content Updates

#### Live Regions for Cart Updates
```html
<!-- Status updates for screen readers -->
<div aria-live="polite" aria-atomic="true" class="sr-only" id="cart-status">
  <!-- Updates announced by screen readers -->
</div>

<div aria-live="assertive" aria-atomic="true" class="sr-only" id="error-status">
  <!-- Urgent updates (errors) announced immediately -->

<script>
window.nextReady.push(function() {
  const cartStatus = document.getElementById('cart-status');
  const errorStatus = document.getElementById('error-status');
  
  // Announce cart updates
  next.on('cart:item-added', function(data) {
    cartStatus.textContent = `Added ${data.name} to cart. Cart now has ${data.cartCount} items.`;
  });
  
  next.on('cart:item-removed', function(data) {
    cartStatus.textContent = `Removed ${data.name} from cart. Cart now has ${data.cartCount} items.`;
  });
  
  next.on('cart:quantity-updated', function(data) {
    cartStatus.textContent = `Updated ${data.name} quantity to ${data.quantity}.`;
  });
  
  // Announce errors
  next.on('error', function(error) {
    errorStatus.textContent = `Error: ${error.message}`;
  });
});
</script>
```

```html
        aria-describedby="add-to-cart-help"

<script>
// Enhanced add to cart with accessibility
function enhanceAddToCart() {
  document.querySelectorAll('[data-next-action="add-to-cart"]').forEach(button => {
    button.addEventListener('click', function() {
      const statusElement = this.querySelector('#add-status');
      statusElement.textContent = 'Adding to cart...';
      
      // Disable button during operation
      this.disabled = true;
      this.setAttribute('aria-disabled', 'true');
      
      // Re-enable after operation
      setTimeout(() => {
        this.disabled = false;
        this.setAttribute('aria-disabled', 'false');
        statusElement.textContent = 'Added to cart';
        
        // Clear status after delay
        setTimeout(() => {
          statusElement.textContent = '';
        }, 3000);
      }, 500);
    });
  });
}

window.nextReady.push(enhanceAddToCart);
</script>
```

```html
<script>
function openCart() {
  const modal = document.getElementById('cart-modal');
  const backdrop = document.querySelector('.modal-backdrop');
  
  // Show modal
  modal.setAttribute('aria-hidden', 'false');
  backdrop.setAttribute('aria-hidden', 'false');
  
  // Focus management
  modal.querySelector('.close-button').focus();
  
  // Trap focus in modal
  trapFocus(modal);
  
  // Prevent body scroll
  document.body.style.overflow = 'hidden';
}

function closeCart() {
  const modal = document.getElementById('cart-modal');
  const backdrop = document.querySelector('.modal-backdrop');
  
  // Hide modal
  modal.setAttribute('aria-hidden', 'true');
  backdrop.setAttribute('aria-hidden', 'true');
  
  // Restore focus to trigger element
  document.querySelector('[data-toggle="cart"]').focus();
  
  // Restore body scroll
  document.body.style.overflow = '';
}

function trapFocus(element) {
  const focusableElements = element.querySelectorAll(
    'button, [href], input, select, textarea, [tabindex]:not([tabindex="-1"])'
  );
  
  const firstElement = focusableElements[0];
  const lastElement = focusableElements[focusableElements.length - 1];
  
  element.addEventListener('keydown', function(e) {
    if (e.key === 'Tab') {
      if (e.shiftKey) {
        if (document.activeElement === firstElement) {
          lastElement.focus();
          e.preventDefault();
        }
      } else {
        if (document.activeElement === lastElement) {
          firstElement.focus();
          e.preventDefault();
        }
      }
    }
    
    if (e.key === 'Escape') {
      closeCart();
    }
  });
}
</script>
```

#### Keyboard Event Handling
```javascript
window.nextReady.push(function() {
  // Enhanced keyboard support for cart operations
  document.addEventListener('keydown', function(e) {
    // Add to cart with Enter or Space
    if (e.target.matches('[data-next-action="add-to-cart"]')) {
      if (e.key === 'Enter' || e.key === ' ') {
        e.preventDefault();
        e.target.click();
      }
    }
    
    // Quantity controls with arrow keys
    if (e.target.matches('.quantity-input')) {
      const currentValue = parseInt(e.target.value);
      
      if (e.key === 'ArrowUp') {
        e.preventDefault();
        e.target.value = currentValue + 1;
        e.target.dispatchEvent(new Event('change'));
      } else if (e.key === 'ArrowDown' && currentValue > 1) {
        e.preventDefault();
        e.target.value = currentValue - 1;
        e.target.dispatchEvent(new Event('change'));
      }
    }
    
    // Close modals with Escape
    if (e.key === 'Escape') {
      const openModal = document.querySelector('[role="dialog"][aria-hidden="false"]');
      if (openModal) {
        closeModal(openModal);
      }
    }
  });
});
```

#### Gesture Support
```javascript
// Enhanced touch interactions
window.nextReady.push(function() {
  // Swipe to remove cart items on mobile
  if ('ontouchstart' in window) {
    let startX, startY, currentX, currentY;
    
    document.addEventListener('touchstart', function(e) {
      if (e.target.closest('.cart-item')) {
        startX = e.touches[0].clientX;
        startY = e.touches[0].clientY;
      }
    });
    
    document.addEventListener('touchmove', function(e) {
      if (startX && e.target.closest('.cart-item')) {
        currentX = e.touches[0].clientX;
        currentY = e.touches[0].clientY;
        
        const deltaX = currentX - startX;
        const deltaY = currentY - startY;
        
        // Horizontal swipe
        if (Math.abs(deltaX) > Math.abs(deltaY) && Math.abs(deltaX) > 50) {
          const cartItem = e.target.closest('.cart-item');
          cartItem.style.transform = `translateX(${deltaX}px)`;
          
          if (Math.abs(deltaX) > 100) {
            cartItem.classList.add('swipe-to-remove');
          } else {
            cartItem.classList.remove('swipe-to-remove');
          }
        }
      }
    });
    
    document.addEventListener('touchend', function(e) {
      if (startX && e.target.closest('.cart-item')) {
        const cartItem = e.target.closest('.cart-item');
        
        if (cartItem.classList.contains('swipe-to-remove')) {
          // Confirm removal
          if (confirm('Remove this item from cart?')) {
            const itemId = cartItem.dataset.itemId;
            next.removeFromCart(itemId);
          }
        }
        
        // Reset
        cartItem.style.transform = '';
        cartItem.classList.remove('swipe-to-remove');
        startX = startY = null;
      }
    });
  }
});
```

### Accessibility Monitoring

#### Real-time Accessibility Checks
```javascript
window.nextReady.push(function() {
  // Monitor for missing alt text
  const observer = new MutationObserver(function(mutations) {
    mutations.forEach(function(mutation) {
      mutation.addedNodes.forEach(function(node) {
        if (node.nodeType === 1) { // Element node
          const images = node.querySelectorAll ? node.querySelectorAll('img') : [];
          images.forEach(function(img) {
            if (!img.alt && !img.getAttribute('aria-hidden')) {
              console.warn('Image missing alt text:', img);
            }
          });
        }
      });
    });
  });
  
  observer.observe(document.body, {
    childList: true,
    subtree: true
  });
  
  // Monitor form labels
  next.on('form:field-added', function(field) {
    const input = document.getElementById(field.id);
    const label = document.querySelector(`label[for="${field.id}"]`);
    
    if (!label && !input.getAttribute('aria-label')) {
      console.warn('Form field missing label:', input);
    }
  });
});
```
